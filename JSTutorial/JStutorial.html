      //Las funciones y acciones en JS son como

      //Funcion que devuelve el factorial de 
        //un numero recursivamente.
      //De paso hacemos una deteccione de 
        //errores con el bloque try/catch.
      function factorial(i)
      {
      	try
      	{
      	  if(i==0 || i==undefined || isNaN(i))
      	  return "Tiene que ser un numero!";
     	  if(i==1)
      	    return i;
      	  else
      	    return i*factorial(i-1);
      	}
      	catch(err)
      	{
      	  return	err.message;
      	}
      }
      //funcion que se llama al clickar el boton:
      function fact()
      {
      	document.getElementById("num").value = 
      	  factorial(document.getElementById("num").value);
      }

      //Tambien podemos hacer una accion en JS:
      function escribe()
      {
      	document.write("Pues esto de las funciones
      	  en Js funciona");
      }
      escribe();
    </script>
    <input id="num" type="text"></input>
	<button class="btn btn-info" onclick="fact()">Factorial!</button>				
  </body>

<html>
  <head></head>

    <script type="text/javascript">
    //funcion que genera el n valor en la serie de fibonacci
    function fibonacci(n)
    {
      if(n<2)
        return n;
      else 
        return fibonacci(n-2)+fibonacci(n-1);
    }

    //Funcion que escribe la serie de fibonacci desde 0 hasta el numero que el 
    //usuario ha indicado(maximo 20).
    function boto()
    {
      var max = parseInt(document.getElementById("valFibo").value);
      if(max >=0 && max <=20)
      {
        document.getElementById("fibo").innerHTML ="";
        //Aqui tenemos el bucle for:
        for(var i=0;i<=max;i++)
        {
          document.getElementById("fibo").innerHTML += fibonacci(i);
          if(i!=(max))
            document.getElementById("fibo").innerHTML += ",";
        }
        document.getElementById("fibo").innerHTML += ".";
       }
       else
         document.getElementById("fibo").innerHTML ="Posa un numero entre 1 i 20.";
    }
  </script>
  <input id="valFibo" class="inputEx" type="number" min="1" max="20" value="5"/>
  <button class="btn btn-info" onclick="boto()">Fibonacci</button>
  <br/>
  <span id="fibo"></span>
  </body>

<html>

    <script type="text/javascript">
      function busc()
      {
        var frase = document.getElementById("valFrase").value;
        var car = document.getElementById("valBusc").value;
        if(frase.trim() == "")
          document.getElementById("busc").innerHTML = "Introduce una frase donde buscar.";
        else if(car.trim() == "")
          document.getElementById("busc").innerHTML = "Introduce un caracter a buscar.";
        else
        {
          var trobat = false;
          frase +=".";
          var i = 0;

          //Haremos un bucle para buscar un caracter en una cadena de texto
          //El bucle acaba si encontramos el caracter, o llegamos al final
          while(!trobat && frase[i]!='.')
          {
            if(frase[i]==car)
            trobat = true;
            else i++;
          }
          if(trobat)
            document.getElementById("busc").innerHTML = "El caracter esta en la posicion "+(i+1)+" .";
          else
            document.getElementById("busc").innerHTML = "No se ha encontrado el caracter.";
        }
      }
    </script>
    <input id="valFrase" class="inputEx" type="text" cols="4" placeholder="Introduce la frase donde buscar" maxlength="48"/>
    <input id="valBusc" class="inputEx" type="text" placeholder="c" maxlength="1"/>
    <button class="btn btn-info" onclick="busc()">Buscar</button>
    <br/>
    <span id="busc"></span>
  </body>
